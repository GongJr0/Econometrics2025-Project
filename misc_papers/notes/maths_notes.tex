\textbf{Keynesian AD}\footnote{Expected change in [unadjusted] price level equals the expected nominal price inflation.}
\[
E[\pi_{t+1}] = \frac{E_t[P_{t+1}] - P_t}{P_t} 
\]

\[
Y_t = \hat{Y} - \alpha \Bigg(i_t - \frac{E_t[P_{t+1}]}{P_t} + 1\Bigg)
\]

\textbf{Price Elasticity of Output}


\[
\frac{\partial Y_t}{\partial P_t} \cdot \frac{P_t}{Y_t} = -\frac{\alpha E_t[P_{t+1}]}{P^2_{t}}
\]
